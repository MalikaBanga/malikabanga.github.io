Simple JavaScript & SVG Chart Demo

Bar Chart

m('#bar-chart').barChart({
    height: 400,
    width:600,
    left:50,
    bottom:50,
    maxBarVal:140,
    minBarVal:0,
    xAxis: {
        text: 'Year',
        plotValue: [1951,1961,1971,1981,1991,2001,2011],
    },
    yAxis: {
        text: 'Population in Crore',
        plotValueDifference: 20,
        plotValue: [36.11,43.93,54.82,68.33,84.64,102.87,121.07],
    },
});
